/* ════════════════════════════════════════════════════════
   workout-ai.store.js — In-workout AI data layer
   ════════════════════════════════════════════════════════ */

import { DB } from './db.js';
import { State as WorkoutState } from './workout.store.js';

/* ══════════════════════════════════════════════
   AI STATE — shared state for in-workout AI
   ══════════════════════════════════════════════ */

/** @type {boolean} */
let _hasSuggestion = false;
/** @type {string} */
let _suggestionText = '';
/** @type {Array<{role: string, content: string}>} */
let _chatHistory = [];
/** @type {boolean} */
let _streaming = false;

export const WorkoutAIState = {
  get hasSuggestion() { return _hasSuggestion; },
  set hasSuggestion(v) { _hasSuggestion = v; },
  get suggestionText() { return _suggestionText; },
  set suggestionText(v) { _suggestionText = v; },
  get chatHistory() { return _chatHistory; },
  set chatHistory(v) { _chatHistory = v; },
  get isStreaming() { return _streaming; },
  set isStreaming(v) { _streaming = v; },
};

/**
 * Check if AI should show suggestion based on current workout state.
 * Proactive triggers:
 * - First set of exercise → form tip
 * - Weight increase → confidence boost
 * - 3+ sets at same weight → plateau nudge
 * - High fatigue (>60%) → deload suggestion
 * @param {Object} workout — current active workout
 * @param {Object} fatigue — muscle fatigue scores
 * @returns {Promise<boolean>}
 */
export async function checkProactiveSuggestion(workout, fatigue) {
  if (!workout || workout.phase !== 'active') {
    _hasSuggestion = false;
    return false;
  }

  const plan = workout.plan || [];
  let shouldShow = false;
  let suggestion = '';

  // Check for first set of new exercise
  for (const ex of plan) {
    const doneSets = (ex.sets || []).filter(s => s.done).length;
    if (doneSets === 0) {
      shouldShow = true;
      suggestion = `💪 Starting **${ex.name}** today! Focus on controlled tempo — 3s down, 1s pause, explosive up.`;
      break;
    }
  }

  // Check for plateau (same weight 3+ sessions)
  if (!shouldShow) {
    const history = await DB.Workouts.getAll();
    const sameExerciseHistory = history.filter(w =>
      (w.exercises || []).some(e => e.name === plan[0]?.name)
    );

    if (sameExerciseHistory.length >= 3) {
      const lastWeights = sameExerciseHistory.slice(-3).map(w => {
        const ex = w.exercises.find(e => e.name === plan[0]?.name);
        return ex?.sets?.[0]?.weight || 0;
      });

      const allSame = lastWeights.every((w, _, arr) => w === arr[0]);
      if (allSame) {
        shouldShow = true;
        suggestion = `📊 You've hit **${plan[0]?.name}** at the same weight for 3 sessions. Want to try a deload or technique variation?`;
      }
    }
  }

  // Check for high fatigue
  const highFatigueMuscles = Object.entries(fatigue).filter(([, v]) => v > 0.6);
  if (highFatigueMuscles.length > 0 && !shouldShow) {
    shouldShow = true;
    const muscles = highFatigueMuscles.map(([m]) => m).join(', ');
    suggestion = `⚠️ High fatigue detected in **${muscles}**. Consider reducing volume by 20% or extending rest to 3-4min.`;
  }

  _hasSuggestion = shouldShow;
  _suggestionText = suggestion;

  return shouldShow;
}

/**
 * Send message to AI coach and stream response.
 * @param {string} message — user message
 * @param {Object} opts — { onText, onDone, onError }
 * @returns {Promise<void>}
 */
export async function sendMessage(message, { onText, onDone, onError }) {
  if (_streaming) return;
  _streaming = true;

  // Build context from current workout
  const ctx = {
    workout: WorkoutState,
    exercises: WorkoutState.plan || []
  };

  try {
    const response = await fetch('/api/coach', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        workouts: (await DB.Workouts.getAll()).slice(0, 3).map(w => ({
          type: w.type,
          hoursAgo: Math.round((Date.now() - w.timestamp) / 3600000),
          tonnageKg: Math.round(w.tonnage || 0),
          exercises: (w.exercises || []).slice(0, 3).map(e => ({
            name: e.name,
            sets: (e.sets || []).filter(s => s.done).length
          }))
        })),
        topLifts: (await DB.OneRM.getAll()).slice(0, 3).map(o => ({
          exercise: o.id,
          oneRM: o.value
        })),
        messages: [..._chatHistory, { role: 'user', content: message }]
      })
    });

    if (!response.ok) {
      const err = await response.json().catch(() => ({}));
      throw new Error(err.error || `HTTP ${response.status}`);
    }

    // Read SSE stream
    const reader = response.body.getReader();
    const decoder = new TextDecoder();
    let buffer = '';
    let aiText = '';

    outer: while (true) {
      const { done, value } = await reader.read();
      if (done) break;
      buffer += decoder.decode(value, { stream: true });

      const lines = buffer.split('\n');
      buffer = lines.pop() ?? '';

      for (const line of lines) {
        if (!line.startsWith('data: ')) continue;
        const raw = line.slice(6).trim();
        if (raw === '[DONE]') break outer;
        try {
          const { text, error } = JSON.parse(raw);
          if (error) throw new Error(error);
          if (text) {
            aiText += text;
            onText(text);
          }
        } catch (parseErr) {
          if (parseErr.message !== 'Unexpected end of JSON input') throw parseErr;
        }
      }
    }

    // Update history
    _chatHistory.push({ role: 'user', content: message });
    _chatHistory.push({ role: 'assistant', content: aiText });

    onDone(aiText);
  } catch (err) {
    onError(err.message);
  } finally {
    _streaming = false;
  }
}

// @ts-check
/* ════════════════════════════════════════════════════════
   workout-ai.view.js — In-workout AI rendering
   ════════════════════════════════════════════════════════ */

import { WorkoutAIState, checkProactiveSuggestion, sendMessage, QUICK_QUESTIONS } from './workout-ai.store.js';
import { State as WorkoutState } from './workout.store.js';
import { Heatmap } from './claude.store.js';

/* ══════════════════════════════════════════════
   INITIALIZATION
   ══════════════════════════════════════════════ */

/**
 * Initialize in-workout AI — check for proactive suggestions.
 */
export async function init() {
  // Load fatigue scores
  const scores = await Heatmap.compute();

  // Check if AI should show suggestion
  const shouldShow = await checkProactiveSuggestion(WorkoutState, scores);

  // Update bubble visibility
  const bubble = document.getElementById('workout-ai-bubble');
  if (bubble) {
    bubble.hidden = !shouldShow;
  }

  // Store suggestion for display
  if (shouldShow) {
    console.log('[WorkoutAI] Proactive suggestion available');
  }
}

/* ══════════════════════════════════════════════
   BUBBLE VISIBILITY
   ══════════════════════════════════════════════ */

/**
 * Update bubble visibility based on current state.
 */
export function updateBubble() {
  const bubble = document.getElementById('workout-ai-bubble');
  if (!bubble) return;

  bubble.hidden = !WorkoutAIState.hasSuggestion;
}

/**
 * Show bubble with animation.
 */
export function showBubble() {
  const bubble = document.getElementById('workout-ai-bubble');
  if (bubble) {
    bubble.hidden = false;
    bubble.style.opacity = '1';
    bubble.style.transform = 'scale(1)';
  }
}

/**
 * Hide bubble with animation.
 */
export function hideBubble() {
  const bubble = document.getElementById('workout-ai-bubble');
  if (bubble) {
    bubble.style.opacity = '0';
    bubble.style.transform = 'scale(0.8)';
    setTimeout(() => {
      bubble.hidden = true;
    }, 300);
  }
}

/* ═════════════════════════════════════════════
   CHAT TOGGLE
   ══════════════════════════════════════════════ */

/**
 * Toggle mini chat overlay.
 */
export function toggle() {
  const chat = document.getElementById('workout-ai-chat');
  if (!chat) return;

  const isHidden = chat.hidden;
  chat.hidden = !isHidden;

  if (isHidden) {
    // Opening — render chat
    renderChat();
    // Focus input
    setTimeout(() => {
      document.getElementById('workout-ai-input')?.focus();
    }, 100);
  }
}

/* ══════════════════════════════════════════════
   CHAT RENDERING
   ══════════════════════════════════════════════ */

/**
 * Render chat messages.
 */
export function renderChat() {
  const container = document.getElementById('workout-ai-messages');
  if (!container) return;

  const history = WorkoutAIState.chatHistory;

  if (history.length === 0) {
    // Show proactive suggestion if available
    if (WorkoutAIState.suggestionText) {
      container.innerHTML = `
        <div class="chat-message ai">
          ${markdownToHtml(WorkoutAIState.suggestionText)}
        </div>
      `;
    } else {
      container.innerHTML = `
        <div class="chat-message ai">
          👋 Hey! I'm your Coach. Ask me anything about this workout!
        </div>
      `;
    }
  } else {
    container.innerHTML = history.map(msg => `
      <div class="chat-message ${msg.role}">
        ${markdownToHtml(msg.content)}
      </div>
    `).join('');
  }

  // Scroll to bottom
  container.scrollTop = container.scrollHeight;
}

/**
 * Add message to chat.
 * @param {string} text — message text
 * @param {'user'|'ai'} role
 */
export function addMessage(text, role) {
  const container = document.getElementById('workout-ai-messages');
  if (!container) return;

  const div = document.createElement('div');
  div.className = `chat-message ${role}`;
  div.innerHTML = markdownToHtml(text);
  container.appendChild(div);
  container.scrollTop = container.scrollHeight;
}

/* ══════════════════════════════════════════════
   SEND MESSAGE
   ══════════════════════════════════════════════ */

/**
 * Send user message to AI.
 */
export async function send() {
  const input = document.getElementById('workout-ai-input');
  if (!input) return;

  const message = input.value.trim();
  if (!message) return;

  // Clear input
  input.value = '';

  // Show user message
  addMessage(message, 'user');

  // Stream AI response
  await sendMessage(message, {
    onText: (text) => {
      // Streaming — append to last AI message
      let lastAi = document.querySelector('.chat-message.ai:last-child');
      if (!lastAi) {
        lastAi = document.createElement('div');
        lastAi.className = 'chat-message ai';
        document.getElementById('workout-ai-messages')?.appendChild(lastAi);
      }
      lastAi.innerHTML = markdownToHtml(text);
    },
    onDone: (text) => {
      console.log('[WorkoutAI] Response complete:', text);
    },
    onError: (err) => {
      addMessage(`⚠️ Error: ${err}`, 'ai');
    }
  });
}

/**
 * Handle Enter key in input.
 * @param {KeyboardEvent} event
 */
export function handleKey(event) {
  if (event.key === 'Enter') {
    send();
  }
}

/**
 * Quick preset question.
 * @param {keyof typeof QUICK_QUESTIONS} type
 */
export async function quickAsk(type) {
  const question = QUICK_QUESTIONS[type];
  if (!question) return;

  // Add to chat
  addMessage(question, 'user');

  // Send to AI
  await sendMessage(question, {
    onText: (text) => {
      let lastAi = document.querySelector('.chat-message.ai:last-child');
      if (!lastAi) {
        lastAi = document.createElement('div');
        lastAi.className = 'chat-message ai';
        document.getElementById('workout-ai-messages')?.appendChild(lastAi);
      }
      lastAi.innerHTML = markdownToHtml(text);
    },
    onDone: () => {},
    onError: (err) => {
      addMessage(`⚠️ Error: ${err}`, 'ai');
    }
  });
}

/* ══════════════════════════════════════════════
   UTILS
   ══════════════════════════════════════════════ */

/**
 * Simple markdown to HTML (bold only).
 * @param {string} text
 * @returns {string}
 */
function markdownToHtml(text) {
  return text
    .replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>')
    .replace(/\*(.*?)\*/g, '<em>$1</em>')
    .replace(/\n/g, '<br>');
}
