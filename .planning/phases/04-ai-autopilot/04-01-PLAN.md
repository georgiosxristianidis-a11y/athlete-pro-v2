### Task 1: Create `/api/generate-plan` endpoint

**File:** `routes/coach.js`

**Implementation:**
```javascript
router.post('/generate-plan', async (req, res) => {
  const {
    workoutHistory = [],    // Last 10 workouts
    oneRMs = [],            // Top 1RM estimates
    goals = 'strength',     // 'strength' | 'hypertrophy' | 'endurance'
    experience = 'intermediate' // 'beginner' | 'intermediate' | 'advanced'
  } = req.body;

  // Validate
  if (!workoutHistory.length && !goals) {
    // New user — return DEFAULT_PLAN structure
    return res.json({
      success: true,
      plan: DEFAULT_PLAN,
      generated: true,
      note: 'Default plan — customize based on your goals'
    });
  }

  // Build prompt for Claude
  const system = _buildPlanGenerationPrompt(workoutHistory, oneRMs, goals, experience);

  // Call Claude with tool_use for structured JSON
  const response = await anthropic.messages.create({
    model: 'claude-opus-4-6',
    max_tokens: 1500,
    system,
    messages: [{
      role: 'user',
      content: 'Generate a personalized PPL program. Return JSON only.'
    }],
    tools: [{
      name: 'generate_ppl_program',
      description: 'Generate a structured PPL workout program',
      input_schema: {
        type: 'object',
        properties: {
          push: {
            type: 'array',
            items: {
              type: 'object',
              properties: {
                name: { type: 'string' },
                sets: { type: 'number' },
                reps: { type: 'number' },
                weight: { type: 'number' },
                notes: { type: 'string' }
              },
              required: ['name', 'sets', 'reps', 'weight']
            }
          },
          pull: { /* same schema */ },
          legs: { /* same schema */ }
        },
        required: ['push', 'pull', 'legs']
      }
    }]
  });

  // Extract tool_use result
  const toolResult = response.content.find(c => c.type === 'tool_use');
  if (toolResult?.input) {
    return res.json({
      success: true,
      plan: toolResult.input,
      generated: true,
      aiNotes: response.content[0]?.text
    });
  }

  // Fallback: parse JSON from text
  const plan = _parsePlanFromText(response.content[0]?.text);
  res.json({ success: true, plan, generated: true });
});
```

/**
 * Check if user needs program generation (new user without custom plan).
 * @returns {boolean}
 */
export function needsProgramGeneration() {
  const hasCustomPlan = localStorage.getItem(PLAN_KEY);
  const hasWorkoutHistory = false; // Will check DB in app.js

  return !hasCustomPlan;
}

/**
 * Fetch AI-generated program from /api/generate-plan.
 * @param {Object} options — { workoutHistory, oneRMs, goals, experience }
 * @returns {Promise<Object>} — { push:[], pull:[], legs:[] }
 */
export async function fetchGeneratedPlan(options = {}) {
  const response = await fetch('/api/generate-plan', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      workoutHistory: options.workoutHistory || [],
      oneRMs: options.oneRMs || [],
      goals: options.goals || 'strength',
      experience: options.experience || 'intermediate'
    })
  });

  if (!response.ok) {
    const err = await response.json().catch(() => ({}));
    throw new Error(err.error || `HTTP ${response.status}`);
  }

  const data = await response.json();
  return data.plan;
}
```

2. Check on app init in `app.js`:

// After DOMContentLoaded
async function _checkNewUserOnboarding() {
  const needsPlan = WorkoutStore.needsProgramGeneration();
  if (!needsPlan) return;

  // Load context from DB
  const [workoutHistory, oneRMs] = await Promise.all([
    DB.Workouts.getAll(),
    DB.OneRM.getAll()
  ]);

  // Generate plan silently in background
  try {
    const plan = await WorkoutStore.fetchGeneratedPlan({
      workoutHistory: workoutHistory.slice(0, 10),
      oneRMs: oneRMs.slice(0, 5)
    });

    // Auto-accept for new users (no history)
    if (!workoutHistory.length) {
      WorkoutStore.savePlan(plan);
      Toast.show('🎉 Welcome! Your personalized PPL plan is ready', 'success');
    } else {
      // Show preview for users with history
      _showPlanPreview(plan);
    }
  } catch (err) {
    console.warn('[_checkNewUserOnboarding] Generation failed:', err);
    // Silently use DEFAULT_PLAN
  }
}
```

/**
 * Show plan preview modal.
 * @param {Object} plan — { push:[], pull:[], legs:[] }

export function showPlanPreview(plan) {
  const container = document.getElementById('plan-preview-content');

  document.getElementById('plan-preview-modal').hidden = false;

/**
 * Hide plan preview modal.
 */
export function hidePlanPreview() {
  document.getElementById('plan-preview-modal').hidden = true;
}

/**
 * Accept generated plan and save to localStorage.
 */
export async function acceptGeneratedPlan() {
  // Get plan from preview (stored in ClaudeState)
  const plan = ClaudeState.generatedPlan;
  if (!plan) return;

  // Save to localStorage
  localStorage.setItem(PLAN_KEY, JSON.stringify(plan));

  // Update UI
  hidePlanPreview();
  Toast.show('✅ Plan saved! Start your first session now.', 'success');

  // Refresh workout selector
  if (window.WorkoutView?.refreshSelector) {
    WorkoutView.refreshSelector();
  }
}

/**
 * Request plan regeneration via AI chat.
 */
export async function regeneratePlan() {
  // Open AI chat with pre-filled context
  ClaudeState.isOpen = true;
  ClaudeState.chatHistory = [
    {
      role: 'assistant',
      content: 'I\'ve generated a PPL plan for you. What would you like to adjust?\n\n- Change exercise selection\n- Modify volume (sets/reps)\n- Adjust for specific goals (strength/hypertrophy)\n- Accommodate equipment limitations\n\nJust tell me what to change!'
    }
  ];

  // Re-render chat
  if (window.ClaudeView?.renderChat) {
    ClaudeView.renderChat();
  }

  hidePlanPreview();
}
```

/**
 * Check if regenerate button should be shown.
 */
export function updateRegenerateButton() {
  const btn = document.getElementById('regenerate-plan-btn');
  if (!btn) return;

  const hasPlan = localStorage.getItem(PLAN_KEY);
  btn.hidden = !hasPlan;
}

/**
 * Request plan regeneration via AI.
 */
export async function requestPlanRegeneration() {
  // Load current plan
  const currentPlan = JSON.parse(localStorage.getItem(PLAN_KEY) || '{}');

  // Load context
  const [workoutHistory, oneRMs] = await Promise.all([
    DB.Workouts.getAll(),
    DB.OneRM.getAll()
  ]);

  // Store in ClaudeState for AI context
  ClaudeState.generatedPlan = currentPlan;
  ClaudeState.regenerationContext = {
    workoutHistory,
    oneRMs,
    currentPlan
  };

  // Open AI chat with regeneration prompt
  ClaudeState.isOpen = true;
  ClaudeState.chatHistory = [
    {
      role: 'user',
      content: `Regenerate my PPL plan. Current plan: ${JSON.stringify(currentPlan, null, 2)}.
                Workout history: ${workoutHistory.length} sessions.
                Top 1RMs: ${oneRMs.map(o => `${o.exercise} ${o.value}kg`).join(', ')}.
                What should I change?`
    }
  ];

  renderChat();
}
```
