# Plan 04: Analytics Store/View Split + JSDoc

## Objective
Split `js/analytics.js` (705 lines) into `js/analytics.store.js` (calendar state, data fetching, pure formatters) and `js/analytics.view.js` (DOM rendering, canvas charts, event handlers). Add JSDoc annotations to all public APIs. After this plan, the original `js/analytics.js` file is deleted and the two new files are imported by `js/app.js`.

/**
 * Move calendar to previous month.
 * @returns {{ year: number, month: number }}
 */
export function calPrev() {
  _calMonth--;
  if (_calMonth < 0) { _calMonth = 11; _calYear--; }
  return { year: _calYear, month: _calMonth };
}

/**
 * Move calendar to next month.
 * @returns {{ year: number, month: number }}

  return { year: _calYear, month: _calMonth };
}

4. Create data fetching functions that wrap DB calls:

/**
 * Fetch all analytics data in one call.
 * @returns {Promise<{ workouts: import('./db.js').WorkoutRecord[], orms: import('./db.js').OneRMRecord[], metrics: import('./db.js').MetricsRecord[] }>}
 */
export async function fetchAllData() {
  const [workouts, orms, metrics] = await Promise.all([
    DB.Workouts.getAll(),
    DB.OneRM.getAll(),
    DB.Metrics.getAll(),
  ]);
  _calWorkouts = workouts;
  return { workouts, orms, metrics };
}

/**

}
